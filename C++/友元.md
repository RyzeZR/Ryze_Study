# 友元
## 代码示例
```
#include <iostream>
using namespace std;
class Time
{
public:
    Time(int,int,int);
    friend void display(Time &);
private:
    int hours;
    int minute;
    int sec;
};
Time::Time(int h,int m,int s)
{
    hours = h;
    minute = m;
    sec = s;
}
void display(Time &t)
{
    cout<<t.hours<<":"<<t.minute<<":"<<t.sec<<endl;
}
int main()
{
    Time t1(10,20,56);
    display(t1);
    return 0;
}

```
